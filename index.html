<script>

    const USER_NAME = "リカ"; 

    const USER_AVATAR = "🌷"; 


    let currentWeatherData = "天気情報なし";


    document.addEventListener("DOMContentLoaded", () => {

        loadPosts();

        getLiveWeatherAndLocation();

    });


    function getLiveWeatherAndLocation() {

        if (!navigator.geolocation) {

            document.getElementById("weatherDisplay").innerText = "位置情報非対応";

            return;

        }


        navigator.geolocation.getCurrentPosition(async (position) => {

            const lat = position.coords.latitude;

            const lon = position.coords.longitude;


            try {

                const weatherRes = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current_weather=true`);

                const weatherData = await weatherRes.json();

                const temp = Math.round(weatherData.current_weather.temperature);

                const weatherCode = weatherData.current_weather.weathercode;

                const weatherText = translateWeatherCode(weatherCode);


                const geoRes = await fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lon}&accept-language=ja`);

                const geoData = await geoRes.json();

                

                const address = geoData.address;

                const locationName = address.city || address.town || address.suburb || address.state || "現在地";


                currentWeatherData = `${weatherText} ${temp}°C / ${locationName}`;

                document.getElementById("weatherDisplay").innerText = currentWeatherData;


            } catch (error) {

                console.error(error);

                document.getElementById("weatherDisplay").innerText = "データの取得に失敗しました";

            }

        }, () => {

            document.getElementById("weatherDisplay").innerText = "位置情報の許可が必要です";

        });

    }


    function translateWeatherCode(code) {

        if (code === 0) return "晴れ";

        if ([1, 2, 3].includes(code)) return "晴れのちくもり";

        if ([45, 48].includes(code)) return "霧";

        if ([51, 53, 55, 56, 57, 61, 63, 65, 66, 67, 80, 81, 82].includes(code)) return "雨";

        if ([71, 73, 75, 77, 85, 86].includes(code)) return "雪";

        if ([95, 96, 99].includes(code)) return "雷雨";

        return "くもり";

    }


    function addPost() {

        const content = document.getElementById("postInput").value.trim();

        if (!content) return;


        const now = new Date();

        const timestamp = `${String(now.getMonth() + 1).padStart(2, '0')}/${String(now.getDate()).padStart(2, '0')} ${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`;


        const post = {

            id: Date.now(),

            time: timestamp,

            weather: currentWeatherData, 

            content: content,

            correction: ""

        };


        let posts = JSON.parse(localStorage.getItem("jp_sns_posts")) || [];

        posts.unshift(post);

        localStorage.setItem("jp_sns_posts", JSON.stringify(posts));


        document.getElementById("postInput").value = "";

        renderPosts();

    }


    function renderPosts() {

        const posts = JSON.parse(localStorage.getItem("jp_sns_posts")) || [];

        const feed = document.getElementById("postFeed");

        feed.innerHTML = "";


        posts.forEach(post => {

            const postEl = document.createElement("div");

            postEl.className = "post";

            

            const promptText = `請幫我批改這段日文日記。請指出不自然的地方、文法錯誤，並提供修正建議與更道地的表達方式：\n\n「${post.content}」`;

            const weatherString = post.weather ? `<span>${escapeHtml(post.weather)}</span>` : "";


            postEl.innerHTML = `

                <div class="post-header">

                    <div class="avatar">

                        <span>${escapeHtml(USER_AVATAR)}</span>

                    </div>

                    <div class="post-info">

                        <span class="username">${escapeHtml(USER_NAME)}</span>

                        <div class="meta-info">

                            <span>${post.time}</span>

                            ${weatherString}

                        </div>

                    </div>

                </div>

                <div class="post-content">${escapeHtml(post.content)}</div>

                

                <div class="action-group">

                    <button class="action-btn btn-gemini" onclick="goToAI('gemini', \`${encodeURIComponent(promptText)}\`)">自然な表現に (Gemini)</button>

                    <button class="action-btn btn-claude" onclick="goToAI('claude', \`${encodeURIComponent(promptText)}\`)">自然な表現に (Claude)</button>

                </div>


                <div class="correction-box">

                    <div class="correction-title">添削・学習メモ</div>

                    <div class="correction-content" contenteditable="true" onblur="saveCorrection(${post.id}, this.innerText)" placeholder="タップして添削結果やメモを記入...">

                        ${post.correction || ""}

                    </div>

                </div>

                <button class="delete-btn" onclick="deletePost(${post.id})">削除する</button>

            `;

            feed.appendChild(postEl);

        });

    }


    function goToAI(aiType, encodedText) {

        const text = decodeURIComponent(encodedText);

        let targetUrl = "https://gemini.google.com/";

        let aiName = "Gemini";


        if (aiType === 'claude') {

            targetUrl = "https://claude.ai/";

            aiName = "Claude";

        }


        navigator.clipboard.writeText(text).then(() => {

            alert(`添削プロンプトをコピーしました。\n${aiName}を開きますので、そのまま貼り付けて送信してください。`);

            window.open(targetUrl, "_blank");

        }).catch(err => {

            window.open(targetUrl, "_blank");

        });

    }


    function saveCorrection(postId, text) {

        let posts = JSON.parse(localStorage.getItem("jp_sns_posts")) || [];

        const index = posts.findIndex(p => p.id === postId);

        if (index !== -1) {

            posts[index].correction = text;

            localStorage.setItem("jp_sns_posts", JSON.stringify(posts));

        }

    }


    function deletePost(postId) {

        if(confirm("この日記を削除してもよろしいですか？")) {

            let posts = JSON.parse(localStorage.getItem("jp_sns_posts")) || [];

            posts = posts.filter(p => p.id !== postId);

            localStorage.setItem("jp_sns_posts", JSON.stringify(posts));

            renderPosts();

        }

    }


    function escapeHtml(text) {

        return text

            .replace(/&/g, "&amp;")

            .replace(/</g, "&lt;")

            .replace(/>/g, "&gt;")

            .replace(/"/g, "&quot;")

            .replace(/'/g, "&#039;");

    }

    

    // 雖然在這個精簡版中直接抓取 localStorage，仍預留 loadPosts 以免初始化報錯

    function loadPosts() {

        renderPosts();

    }
